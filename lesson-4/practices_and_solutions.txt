CREATE TABLE sales
(
    id           SERIAL PRIMARY KEY,
    product_name varchar(100),
    category     varchar(50),
    price        numeric(6, 2),
    quantity     int,
    sale_date    date
);

*** Aggregate Practice Questions ***

Find the total sold quantity from the sales table.
Find the average price from the sales table.
Write a query to find how many records there are in each category.
Write a query to find the total quantity for each category.
Write a query to find the average price for each category.
Write a query to find the minimum price for each category.
Write a query to find the maximum price for each category.
Write a query to find how many records there are for each sale_date.
Write a query to find the total quantity for each sale_date.
Write a query to show only the categories whose total quantity is greater than 5.
Show only the categories whose average price is greater than 100.
Write a query to find the total value of price * quantity for each category.
Write a query to find the category with the highest total revenue.
Group by category and show only the categories whose number of records is greater than 2.



------------- Before checking your answers, you must solve these problems. -------------

SELECT SUM(quantity) AS total_quantity
FROM sales;

SELECT AVG(price) AS average_price
FROM sales;

SELECT category, COUNT(*) AS total_records
FROM sales
GROUP BY category;

SELECT category, SUM(quantity) AS total_quantity
FROM sales
GROUP BY category;

SELECT category, AVG(price) AS average_price
FROM sales
GROUP BY category;

SELECT category, MIN(price) AS min_price
FROM sales
GROUP BY category;

SELECT category, MAX(price) AS max_price
FROM sales
GROUP BY category;

SELECT sale_date, COUNT(*) AS total_records
FROM sales
GROUP BY sale_date;

SELECT sale_date, SUM(quantity) AS total_quantity
FROM sales
GROUP BY sale_date;

SELECT category, SUM(quantity) AS total_quantity
FROM sales
GROUP BY category
HAVING SUM(quantity) > 5;

SELECT category
FROM sales
GROUP BY category
HAVING AVG(price) > 100;

SELECT category, SUM(price * quantity) AS total_revenue
FROM sales
GROUP BY category;

SELECT category, SUM(price * quantity) AS total_revenue
FROM sales
GROUP BY category
ORDER BY total_revenue DESC
LIMIT 1;

SELECT category, COUNT() AS total_records
FROM sales
GROUP BY category
HAVING COUNT() > 2;
